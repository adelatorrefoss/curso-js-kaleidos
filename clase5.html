function comparador(originalName) {
   return function (name) {
       if (name.length > originalName.length) {
           return name + ' tiene más letras que ' + originalName
       } else {
           return name + ' tiene menos letras que ' + originalName
       }
   }
}

   var str = " characters";
   var fns = [];

   for(var i = 0; i < list.length; i++) {
       (function (text) {
           fns.push(function(withCharText) {
               if (withCharText) {
                   console.log(text + str);
               } else {
                   console.log(text);
               }
           });
       }(list[i]));
   }

   return fns;
}

    var result = [];

         result.push(fns[i](init));

var add = function (a, b) {
    return a + b;
};

var sub = function (a, b) {
    return a - b;
};

function Calculator(init) {
   return function () {
       var operaciones = arguments;

       return function () {
           var current = init;
           for(var i = 0; i < arguments.length; i++) {
               current = operaciones[i](current, arguments[i]);
           }

           return current;
       }
   }
}

var List = (function() {
    var list = [];

    function find(val) {
        return list.indexOf(val);
    }

    return {
        add: function(val) {
            list.push(val);
        },
        remove: function(val) {
            list.splice(val, 1);
        },
        print: function() {
            console.log(list);
        },
        reset: function() {
            list = [];
        },
        find: function(val) {
            var result = find(val);

            if (result === -1) {
                console.log("Not found");
            } else {
                console.log("Index " + result);
            }
        },
        removeItem: function(val) {
            var result = find(val);

            list.splice(result, 1);
        }
    }
}());

                return obj;
